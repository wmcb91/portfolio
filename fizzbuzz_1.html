const bitwiseFizzBuzz = (max) => {

  const words = ['Fizz', 'Buzz', 'FizzBuzz'];

  let mask = 810092048; // 11 00 00 01 00 10 01 00 00 01 10 00 01 00 00

  let c = 0;

  let output;

  for (let i = 1; i <= max; i++) {

    c = mask & 3; // c = 0

    mask = mask >> 2 | c << 28; // mask = 810092048

    output = words[c - 1] || i; // output = undefined

    console.log(output);

  }

};


bitwiseFizzBuzz(100);
